\chapter{方程建立}

\begin{theorem}{Maxwell 方程组}
    \begin{gather}
        \nabla\times \vec{E}=-\frac{\partial \vec{B}}{\partial t}\\
        \nabla\times \vec{H}=\vec{J} +\varepsilon_0\frac{\partial \vec{E}}{\partial t}\\
        \nabla\cdot \vec{D}=\rho_0\\
        \nabla\cdot \vec{B}=0
    \end{gather}
\end{theorem}
